#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n), freq(n+1, 0);
        for (int &x : a) cin >> x;

        for (int x : a) freq[x]++;

        int ans = 0;
        for (int c = 1; c <= n; c++) {
            int cnt = 0;
            for (int x = 1; x <= n; x++) {
                if (freq[x] >= c) cnt++;
            }
            ans = max(ans, cnt * c);
        }

        cout << ans << endl;
    }
    return 0;
}
