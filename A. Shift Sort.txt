#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        string s;
        cin >> s;

        string sorted = s;
        sort(sorted.begin(), sorted.end());

        int cnt = 0;


        for (int i = 0; i < n; i++) {
            if (s[i] == '1' && sorted[i] == '0') {
                cnt++;
            }
        }
        cout << (cnt + 1) / 2 << endl;
    }

    return 0;
}
