#include <iostream>
#include <iomanip> // for setprecision
using namespace std;

int main() {
    string item;
    int qty;
    double price;

    // ===== INPUT =====
    cin >> item >> qty >> price;

    // ===== PROCESSING =====
    double total = qty * price;

    // ===== OUTPUT =====
    cout << fixed << setprecision(2);
    cout << "Total: " << total << endl;

    return 0;
}
