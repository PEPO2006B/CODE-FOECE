#include <iostream>
#include <string>
using namespace std;

int main() {
    int n;
    cin >> n;

    for(int i = 0; i < n; i++){
        string id, type;
        int entry, exit;
        cin >> id >> type >> entry >> exit;

        int rate;
        if(type == "Car") rate = 5;
        else if(type == "Bus") rate = 10;
        else if(type == "StaffCar") rate = 3;

        int duration = exit - entry;
        int cost = duration * rate;

        cout << id << " " << duration << " " << cost << endl;
    }

    return 0;
}
