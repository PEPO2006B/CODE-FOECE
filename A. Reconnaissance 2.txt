#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;

    int a[101];
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int minDiff = 1000000;
    int soldier1 = 0, soldier2 = 1;


    for (int i = 0; i < n - 1; i++) {
        int diff = abs(a[i] - a[i + 1]);

        if (diff < minDiff) {
            minDiff = diff;
            soldier1 = i;
            soldier2 = i + 1;
        }
    }

    int lastDiff = abs(a[n - 1] - a[0]);
    if (lastDiff < minDiff) {
        soldier1 = n - 1;
        soldier2 = 0;
    }

    cout << soldier1 + 1 << " " << soldier2 + 1 << endl;

    return 0;
}
