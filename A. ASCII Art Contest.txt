#include <iostream>
#include <algorithm>  
using namespace std;

int main() {
    int g, c, l;
    cin >> g >> c >> l;

    int mx = max({ g, c, l });   
    int mn = min({ g, c, l });  

    if (mx - mn >= 10) {
        cout << "check again";
    }
    else {
        int arr[3] = { g, c, l };
        sort(arr, arr + 3);    
        cout << "final " << arr[1]; 
    }

    return 0;
}
